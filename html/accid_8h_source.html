114  bool operator()(const Accid *first, const Accid *second) const
115  {
116  if (first->GetDrawingY() < second->GetDrawingY()) {
117  return true;
118  }
119  else if (first->GetDrawingY() > second->GetDrawingY()) {
120  return false;
121  }
122  else {
123  // with unissons, natural should always be the last accidental (assuming there is a natural)
124  if ((first->GetAccid() == ACCIDENTAL_WRITTEN_n) || (second->GetAccid() == ACCIDENTAL_WRITTEN_n)) {
125  return (first->GetAccid() != ACCIDENTAL_WRITTEN_n);
126  }
127  return first->GetDrawingY() < second->GetDrawingY();
128  }
129  }
