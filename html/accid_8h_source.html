133 public:

136  bool operator()(const Accid *first, const Accid *second) const
137  {
138  if (first->GetDrawingY() < second->GetDrawingY()) {
139  return true;
140  }
141  else if (first->GetDrawingY() > second->GetDrawingY()) {
142  return false;
143  }
144  else {
145  // with unissons, natural should always be the last accidental (assuming there is a natural)
146  if ((first->GetAccid() == ACCIDENTAL_WRITTEN_n) || (second->GetAccid() == ACCIDENTAL_WRITTEN_n)) {
147  return (first->GetAccid() != ACCIDENTAL_WRITTEN_n);
148  }
149  return first->GetDrawingY() < second->GetDrawingY();
150  }
151  }
152 };
