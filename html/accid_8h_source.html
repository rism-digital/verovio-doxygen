157class AccidSpaceSort {
158
159public:
160 AccidSpaceSort() {}
161
162 bool operator()(const Accid *first, const Accid *second) const
163 {
164 if (first->GetDrawingY() < second->GetDrawingY()) {
165 return true;
166 }
167 else if (first->GetDrawingY() > second->GetDrawingY()) {
168 return false;
169 }
170 else {
171 // with unissons, natural should always be the last accidental (assuming there is a natural)
172 if ((first->GetAccid() == ACCIDENTAL_WRITTEN_n) || (second->GetAccid() == ACCIDENTAL_WRITTEN_n)) {
173 return (first->GetAccid() != ACCIDENTAL_WRITTEN_n);
174 }
175 return first->GetDrawingY() < second->GetDrawingY();
176 }
177 }
178};
