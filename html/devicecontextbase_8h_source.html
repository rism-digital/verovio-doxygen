193class Point {
194public:
195 int x, y;
196
197 Point() : x(0), y(0) {}
198 Point(int xx, int yy) : x(xx), y(yy) {}
199
200 // no copy ctor or assignment operator - the defaults are ok
201
202 // comparison
203 bool operator==(const Point &p) const { return x == p.x && y == p.y; }
204 bool operator!=(const Point &p) const { return !(*this == p); }
205
206 // arithmetic operations (component wise)
207 Point operator+(const Point &p) const { return Point(x + p.x, y + p.y); }
208 Point operator-(const Point &p) const { return Point(x - p.x, y - p.y); }
209
210 Point &operator+=(const Point &p)
211 {
212 x += p.x;
213 y += p.y;
214 return *this;
215 }
216 Point &operator-=(const Point &p)
217 {
218 x -= p.x;
219 y -= p.y;
220 return *this;
221 }
222
223 Point operator-() const { return Point(-x, -y); }
224
225 Point min(const Point &p) const
226 {
227 int x = std::min(this->x, p.x);
228 int y = std::min(this->y, p.y);
229 return Point(x, y);
230 }
231
232 Point max(const Point &p) const
233 {
234 int x = std::max(this->x, p.x);
235 int y = std::max(this->y, p.y);
236 return Point(x, y);
237 }
238};
