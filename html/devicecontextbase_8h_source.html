207  bool operator==(const Point &p) const { return x == p.x && y == p.y; }
208  bool operator!=(const Point &p) const { return !(*this == p); }
209 
210  // arithmetic operations (component wise)
211  Point operator+(const Point &p) const { return { x + p.x, y + p.y }; }
212  Point operator-(const Point &p) const { return { x - p.x, y - p.y }; }
213

215  {
216  x += p.x;
217  y += p.y;
218  return *this;
219  }

221  {
222  x -= p.x;
223  y -= p.y;
224  return *this;
225  }
226 
227  Point operator-() const { return { -x, -y }; }
228 
229  Point min(const Point &p) const
230  {
231  int x = std::min(this->x, p.x);
232  int y = std::min(this->y, p.y);
233  return { x, y };
234  }
235 
236  Point max(const Point &p) const
237  {
238  int x = std::max(this->x, p.x);
