219  // acclev == In **mens data, controls the accidental level conversion
220  // from gestural to editorial.
221  // *acclev:0 == no editorial acccidentals visible (all converted to gestural)
222  // *acclev:1 == accidentals marked with YY are displayed as editorial accidentals (above notes)
223  // *acclev:2 == accidentals marked with Y are displayed as editorial accidentals (above notes)
224  // *acclev:3 == accidentals marked with yy are displayed as editorial accidentals (above notes)
225  // *acclev:4 == accidentals marked with y are displayed as editorial accidentals (above notes)
226  // meaning of the accidental marks:
227  // y = algorithmic interpretation of an implicit accidental, primarily used for notes
228  // with accidentals not shown but assigned by the key signature.
229  // yy = a cautionary type of accidental, such as returning to the accidental within the
230  // key signature (such as a b-flat coming after a b-natural in a 1-flat key signature).
231  // Y = a performance interpretation accidental (musica ficta) that is not indicated or
232  // directly implied in the source, but is needed due to performance practice.
233  // YY = an accidental that should be added due to what the editor thinks is an error.
234  //
235  // Equivalences to numbers:
236  // *Xacclev == *acclev:0 all levels are mapped to @accid.ges
237  // *acclev == *acclev:0 all levels are mapped to @accid.ges
238  // *acclev: == *acclev:0 all levels are mapped to @accid.ges
239  // *acclev:YY == *acclev:1 levels 2-4 are mapped to @accid.ges and level 1 to @accd+@edit
240  // *acclev:Y == *acclev:2 levels 3-4 are mapped to @accid.ges and levels 1-2 to @accd+@edit
241  // *acclev:yy == *acclev:3 level 4 is mapped to @accid.ges and levels 1-3 to @accd+@edit
242  // *acclev:y == *acclev:4 all levels are mapped to @accid+@edit
243  //
244  // The default level is *acclev:1 (YY will show as editorial accidental, y, yy, and Y will be @accid.ges)
245  int acclev = 1;
246 
247  // righthalfstem == true means to place half-note stems always on right side
248  // of noteheads. False is standard modern style.
249  bool righthalfstem;
250 
251  // ottavanote == keep track of ottava marks: stores the starting note of

299  // dimensions:

307  int m_dynamstaffadj = 0; // used for grand staff movement of dynamics to lower staff
