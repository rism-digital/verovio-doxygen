97 // To be used with std::sort to find the object with a closest bounding
98 // box to a point defined by the x and y parameters of ClosestBB
99 
100 struct ClosestBB {
101  int x;
102  int y;
103 
104  int distanceToBB(int ulx, int uly, int lrx, int lry, double rotate = 0)
105  {
106  int offset = (x - ulx) * tan(rotate * M_PI / 180.0);
107  uly = uly - offset;
108  lry = lry - offset;
109  int xDiff = std::max((ulx > x ? ulx - x : 0), (x > lrx ? x - lrx : 0));
110  int yDiff = std::max((uly > y ? uly - y : 0), (y > lry ? y - lry : 0));
111 
112  return sqrt(xDiff * xDiff + yDiff * yDiff);
113  }
114 
115  bool operator()(Object *a, Object *b)
116  {
117  if (!a->GetFacsimileInterface() || !b->GetFacsimileInterface()) return true;
118  Zone *zoneA = a->GetFacsimileInterface()->GetZone();
119  Zone *zoneB = b->GetFacsimileInterface()->GetZone();
120 
121  int distA
122  = distanceToBB(zoneA->GetUlx(), zoneA->GetUly(), zoneA->GetLrx(), zoneA->GetLry(), zoneA->GetRotate());
123  int distB
124  = distanceToBB(zoneB->GetUlx(), zoneB->GetUly(), zoneB->GetLrx(), zoneB->GetLry(), zoneB->GetRotate());
125  return (distA < distB);
126  }
127 };

129 // To be used with std::stable_sort to find the position to insert a new staff
130 
131 struct StaffSort {
132  // Sort staves left-to-right and top-to-bottom
133  // Sort by y if there is no intersection, by x if there is
134  bool operator()(Object *a, Object *b)
135  {
136  if (!a->GetFacsimileInterface() || !b->GetFacsimileInterface()) return true;
137  Zone *zoneA = a->GetFacsimileInterface()->GetZone();
138  Zone *zoneB = b->GetFacsimileInterface()->GetZone();
139 
140  int aLowest, bLowest, aHighest, bHighest;
141 
142  aLowest = zoneA->GetRotate() < 0
143  ? zoneA->GetLry()
144  : zoneA->GetLry() + (zoneA->GetLrx() - zoneA->GetUlx()) * tan(zoneA->GetRotate() * M_PI / 180.0);
145 
146  aHighest = zoneA->GetRotate() < 0
147  ? zoneA->GetUly()
148  : zoneA->GetUly() - (zoneA->GetLrx() - zoneA->GetUlx()) * tan(zoneA->GetRotate() * M_PI / 180.0);
149 
150  bLowest = zoneB->GetRotate() < 0
151  ? zoneB->GetLry()
152  : zoneB->GetLry() + (zoneB->GetLrx() - zoneB->GetUlx()) * tan(zoneB->GetRotate() * M_PI / 180.0);
153 
154  bHighest = zoneB->GetRotate() < 0
155  ? zoneB->GetUly()
156  : zoneB->GetUly() - (zoneB->GetLrx() - zoneB->GetUlx()) * tan(zoneB->GetRotate() * M_PI / 180.0);
157 
158  // Check for y intersection
159  if ((aLowest <= bLowest && aLowest >= bHighest) || (aHighest <= bLowest && aHighest >= bHighest)
160  || (bLowest <= aLowest && bLowest >= aHighest) || (bHighest <= aLowest && bHighest >= aHighest)) {
161  // sort by x center
162  return (zoneA->GetUlx() < zoneB->GetUlx());
163  }
164  else { // no intersection
165  return (zoneA->GetUly() < zoneB->GetUly());
166  }
167  }
168 };
