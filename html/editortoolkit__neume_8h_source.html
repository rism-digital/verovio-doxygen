113 // To be used with std::sort to find the object with a closest bounding
114 // box to a point defined by the x and y parameters of ClosestBB
115 
116 struct ClosestBB {
117  int x;
118  int y;
119 
120  int distanceToBB(int ulx, int uly, int lrx, int lry, double rotate = 0)
121  {
122  int offset = (x - ulx) * tan(rotate * M_PI / 180.0);
123  uly = uly + offset;
124  lry = lry + offset;
125  int xDiff = std::max((ulx > x ? ulx - x : 0), (x > lrx ? x - lrx : 0));
126  int yDiff = std::max((uly > y ? uly - y : 0), (y > lry ? y - lry : 0));
127 
128  return sqrt(xDiff * xDiff + yDiff * yDiff);
129  }
130 
131  bool operator()(Object *a, Object *b)
132  {
133  if (!a->GetFacsimileInterface() || !b->GetFacsimileInterface()) return true;
134  Zone *zoneA = a->GetFacsimileInterface()->GetZone();
135  Zone *zoneB = b->GetFacsimileInterface()->GetZone();
136 
137  int distA
138  = distanceToBB(zoneA->GetUlx(), zoneA->GetUly(), zoneA->GetLrx(), zoneA->GetLry(), zoneA->GetRotate());
139  int distB
140  = distanceToBB(zoneB->GetUlx(), zoneB->GetUly(), zoneB->GetLrx(), zoneB->GetLry(), zoneB->GetRotate());
141  return (distA < distB);
142  }
143 };

182 struct StaffSort {
183  // Sort staves left-to-right and top-to-bottom
184  // Sort by y if there is no intersection, by x if there is x intersection is smaller than half length of staff line
185 
186  // Update 2024-04:
187  // Used only in neume lines,
188  // System->(Measure->Staff)
189  // Need to sort Measure to sort staff
190  bool operator()(Object *a, Object *b)
191  {
192  if (!a->Is(SYSTEM) || !b->Is(SYSTEM)) return false;
193  if (!a->FindDescendantByType(MEASURE) || !b->FindDescendantByType(MEASURE)) return false;
194  Measure *measureA = dynamic_cast<Measure *>(a->FindDescendantByType(MEASURE));
195  Measure *measureB = dynamic_cast<Measure *>(b->FindDescendantByType(MEASURE));
196  if (!measureA->IsNeumeLine() || !measureB->IsNeumeLine()) return true;
197  Object *staffA = a->FindDescendantByType(STAFF);
198  Object *staffB = b->FindDescendantByType(STAFF);
199  assert(staffA);
200  assert(staffB);
201  Zone *zoneA = staffA->GetFacsimileInterface()->GetZone();
202  Zone *zoneB = staffB->GetFacsimileInterface()->GetZone();
203  assert(zoneA);
204  assert(zoneB);
205 
206  int aLowest, bLowest, aHighest, bHighest;
207 
208  aLowest = zoneA->GetRotate() < 0
209  ? zoneA->GetLry()
210  : zoneA->GetLry() + (zoneA->GetLrx() - zoneA->GetUlx()) * tan(zoneA->GetRotate() * M_PI / 180.0);
211 
212  aHighest = zoneA->GetRotate() < 0
213  ? zoneA->GetUly()
214  : zoneA->GetUly() - (zoneA->GetLrx() - zoneA->GetUlx()) * tan(zoneA->GetRotate() * M_PI / 180.0);
215 
216  bLowest = zoneB->GetRotate() < 0
217  ? zoneB->GetLry()
218  : zoneB->GetLry() + (zoneB->GetLrx() - zoneB->GetUlx()) * tan(zoneB->GetRotate() * M_PI / 180.0);
219 
220  bHighest = zoneB->GetRotate() < 0
221  ? zoneB->GetUly()
222  : zoneB->GetUly() - (zoneB->GetLrx() - zoneB->GetUlx()) * tan(zoneB->GetRotate() * M_PI / 180.0);
223 
224  // Check for y intersection
225  // if the x intersection part is smaller than half of length of staffA
226  // sort by x coordinate
227  if (((aLowest <= bLowest && aLowest >= bHighest) || (aHighest <= bLowest && aHighest >= bHighest)
228  || (bLowest <= aLowest && bLowest >= aHighest) || (bHighest <= aLowest && bHighest >= aHighest))
229  && (zoneA->GetLrx() - zoneB->GetUlx() <= 0.5 * (zoneA->GetLrx() - zoneA->GetUlx()))) {
230  // sort by x center
231  return (zoneA->GetUlx() < zoneB->GetUlx());
232  }
233  else { // no intersection
234  return (zoneA->GetUly() < zoneB->GetUly());
235  }
236  }
237 };
