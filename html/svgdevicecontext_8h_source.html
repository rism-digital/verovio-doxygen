237  void SetAdditionalAttributes(const std::vector<std::string> &additionalAttributes)
238  {
239  for (std::string s : additionalAttributes) {
240  std::string className = s.substr(0, s.find("@")); // parse <element@attribute>, e.g., "note@pname"
241  std::string attributeName = s.substr(s.find("@") + 1);
242  ClassId classId = ObjectFactory::GetInstance()->GetClassId(className);
243  m_svgAdditionalAttributes.insert({ classId, attributeName });
244  }
245  }

329  // Here we hold references to all different glyphs used so far,
330  // including any glyph for the same code but from different fonts.
331  // They will be added at the end of the file as <defs>.
332  // With multiple font support we need to keep track of:
333  // a) the glyph (to check if is has been already added)
334  // b) the id assigned to glyphs on the (that is has been consumed by the already rendered elements)
335  // To keep things as similar as possible to previous versions we generate ids with as uuuu-ss (where uuuu is the
336  // Smulf code for the glyph and ss the per-session suffix) for most of the cases (single font usage). When the same
337  // glyph has been used from several fonts we use uuuu-n-ss where n indicates the collision count.
338  class GlyphRef {
339  public:
340  GlyphRef(const Glyph *glyph, int count, const std::string &postfix);
341  const Glyph *GetGlyph() const { return m_glyph; };
342  const std::string &GetRefId() const { return m_refId; };
343 
344  private:
345  const Glyph *m_glyph;
346  std::string m_refId;
347  };
