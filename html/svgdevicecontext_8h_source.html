234  void SetAdditionalAttributes(const std::vector<std::string> &additionalAttributes)
235  {
236  for (std::string s : additionalAttributes) {
237  std::string className = s.substr(0, s.find("@")); // parse <element@attribute>, e.g., "note@pname"
238  std::string attributeName = s.substr(s.find("@") + 1);
239  ClassId classId = ObjectFactory::GetInstance()->GetClassId(className);
240  m_svgAdditionalAttributes.insert({ classId, attributeName });
241  }
242  }
