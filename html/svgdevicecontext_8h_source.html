239  void SetAdditionalAttributes(const std::vector<std::string> &additionalAttributes)
240  {
241  for (std::string s : additionalAttributes) {
242  std::string className = s.substr(0, s.find("@")); // parse <element@attribute>, e.g., "note@pname"
243  std::string attributeName = s.substr(s.find("@") + 1);
244  ClassId classId = ObjectFactory::GetInstance()->GetClassId(className);
245  m_svgAdditionalAttributes.insert({ classId, attributeName });
246  }
247  }
