232 void SetAdditionalAttributes(const std::vector<std::string> &additionalAttributes)
233 {
234 for (std::string s : additionalAttributes) {
235 std::string className = s.substr(0, s.find("@")); // parse <element@attribute>, e.g., "note@pname"
236 std::string attributeName = s.substr(s.find("@") + 1);
237 ClassId classId = ObjectFactory::GetInstance()->GetClassId(className);
238 m_svgAdditionalAttributes.insert({ classId, attributeName });
239 }
