448class DiatonicSort {
449
450public:
451 DiatonicSort() {}
452
453 bool operator()(const Object *first, const Object *second) const
454 {
455 const Note *n1 = dynamic_cast<const Note *>(first);
456 const Note *n2 = dynamic_cast<const Note *>(second);
457 assert(n1 && n2);
458 return (n1->GetDiatonicPitch() < n2->GetDiatonicPitch());
459 }
460};
